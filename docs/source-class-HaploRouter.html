 60:         /**

 73:         /**
 74:          * Static helper method used to ensure only one instance of the class is instantiated
 75:          * This overrides the base version in the abstract HaploSingleton class because we 
 76:          * need to support parameters
 77:          *
 78:          * @param array $urls URL pattern mappings t0 process - the first match found will be selected
 79:          * @param string $actionsPath Path to folder which contains actions (by default ../actions relative to index.php)
 80:          * @return HaploRouter
 81:          * @author Ed Eliot
 82:          **/
 83:         public static function get_instance($urls) {
 84:             global $config;
 85:             
 86:             $class = get_called_class();
 87:             
 88:             if (!isset(self::$instances[$class])) {
 89:                 self::$instances[$class] = new $class($urls, $config->get_key('paths', 'actions'));
 90:             }
 91:             return self::$instances[$class];
 92:         }

222:         /**
223:          * Logic to work out which view should be loaded and process parameters
224:          *
225:          * @param array $urls URL patterns to process current URL against
226:          * @return void
227:          * @author Ed Eliot
228:          **/
229:         protected function process($urls) {
230:             // by default assume no match
231:             $match = false;
232:             
233:             // loop through list of URL regex patterns to find a match - 
234:             // processing stops as soon as a match is found so URL patterns 
235:             // should be ordered with the most important first
236:             foreach ($urls as $regEx => $dest) {
237:                 $requestVars = array();
238:                 
239:                 // does the current URL match - if so capture matched sub-groups
240:                 if (preg_match("#^$regEx(\?.*)?$#", $this->get_request_uri(), $this->requestVars)) {
241:                     // the first match is the full URL - we not really 
242:                     // interested in that so drop
243:                     array_shift($requestVars);
244:                     
245:                     // rule could be simple mapping to action or more complex 
246:                     // details containing mapping type etc
247:                     if (is_array($dest)) {
248:                         // if complex form a mapping type must be present (action, redirect, sub-patterns etc)
249:                         if (!empty($dest['type'])) {
250:                             switch ($dest['type']) {
251:                                 // sets name of action in class property
252:                                 case 'action':
253:                                     if (!empty($dest['action'])) {
254:                                         $this->set_action($dest['action'], $this->requestVars);
255:                                     } else {
256:                                         throw new HaploNoActionDefinedException("No action defined for $regEx.");
257:                                     }
258:                                     break;
259:                                 // performs http redirect (301, 302, 404 etc)
260:                                 case 'redirect':
261:                                     if (!empty($dest['url'])) {
262:                                         if (!empty($dest['code'])) {
263:                                             $this->redirect($dest['url'], $dest['code']);
264:                                         } else {
265:                                             $this->redirect($dest['url']);
266:                                         }
267:                                     } else {
268:                                         throw new HaploNoRedirectUrlDefinedException("No redirect URL defined for $regEx.");
269:                                     }
270:                                     break;
271:                                 case 'sub-patterns':
272:                                     // process sub URL patterns - this allows one to create sub pattern matches under 
273:                                     // a base pattern for efficency reasons
274:                                     $subPatterns = array();
275:                                     
276:                                     foreach ($dest['sub-patterns'] as $subRegEx => $subDest) {
277:                                         $subPatterns[$regEx.$subRegEx] = $subDest;
278:                                     }
279:                                     $this->process($subPatterns);
280:                                     break;
281:                                 default:
282:                                     throw new HaploActionTypeNotSupportedException("Action type not supported for $regEx. Should be one of 'action', 'redirect' or 'sub-patterns'.");
283:                             }
284:                         } else {
285:                             throw new HaploNoActionDefinedException("No action type defined for $regEx. Should be one of 'action', 'redirect' or 'sub-patterns'.");
286:                         }
287:                     } else {
288:                         $this->set_action($dest);
289:                     }
290:                     
291:                     $match = true;
292:                     // we don't want to continue processing patterns if a matching one has been found
293:                     break;
294:                 }
295:             }
296:             
297:             // if none of the URL patterns matches look for a default 404 action
298:             if (!$match) {
299:                 // send http 404 error header
300:                 header('HTTP/1.1 404 Not Found');
301:                 
302:                 // check for existence of a 404 action
303:                 if (file_exists("$this->actionsPath/page-not-found.php")) {
304:                     $this->set_action('page-not-found');
305:                 } else {
306:                     throw new HaploNoDefault404DefinedException("No default 404 action found. Add a file named page-not-found.php to $this->actionsPath/ to suppress this message.");
307:                 }
308:             }
309:             
310:             return $match;
311:         }
312:         
313:         /**
314:          *  Simply sets the name of the action to include - it isn't actually 
315:          *  included within this class so that the PHP it contains remains 
316:          *  outside the scope of the class
317:          *
318:          * @param string $action Name of action to set as selected
319:          * @param array $requestVars Details of request variables to try and map to dynamically generated action name
320:          * @return void
321:          * @author Ed Eliot
322:          **/
323:         protected function set_action($action, $requestVars = array()) {
324:             // map request variables to action name if applicable
325:             foreach ($this->requestVars as $key => $value) {
326:                 $action = str_replace("<$key>", $value, $action);
327:             }
328:             
329:             $actionPath = "$this->actionsPath/$action.php";
330:             
331:             if (file_exists($actionPath)) {
332:                 $this->action = $action;
333:                 $this->actionPath = $actionPath;
334:             } else {
335:                 $this->action = '';
336:                 $this->actionPath = '';
337:                 throw new HaploActionNotFoundException("Specified action ($action) hasn't been created.");
338:             }
339:         }
340:         
341:         /**
342:          * Do http redirect (301, 302, 404 etc)
343:          *
344:          * @param string $url URL to redirect to
345:          * @param integer $code HTTP code to send - probably 301 (permanently modified) or 302 (temporarily moved)
346:          * @return void
347:          * @author Ed Eliot
348:          **/
349:         protected function redirect($url, $code = 302) {
350:             // map request variables to URL if applicable
351:             foreach ($this->requestVars as $key => $value) {
352:                 $url = str_replace("<$key>", $value, $url);
353:             }
354:             
355:             header("Location: $url", true, $code);
356:             exit;
357:         }
