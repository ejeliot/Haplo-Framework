 43:     class HaploRss {
 44:         protected $rssObject;
 45:         protected $numItems;
 46:         protected $outputEncoding;
 47:         protected $inItem = false;
 48:         protected $inChannel = false;
 49:         protected $tag = '';
 50:         protected $pointer = -1;
 51:         protected $tempContent = '';
 52:         protected $successful = false;
 53:       
 54:         // $numItems = -1 means whatever's in the feed
 55:         // outputEncoding - defaults to UTF-8
 56:         public function __construct($url, $numItems = -1, $outputEncoding = 'UTF-8') {
 57:             $this->rssObject = new RssObject(); // this object holds the returned data
 58:             $this->numItems = $numItems;
 59:             $this->outputEncoding = $outputEncoding;
 60: 
 61:             // request feed
 62:             if ($data = HaploHttp::get($url)) {
 63:                 $this->successful = $this->parse($data);
 64:             }
 65:         }
 66:       
 67:         // gets the returned feed data structure
 68:         public function get_rss_object() {
 69:             if ($this->successful) {
 70:                 return $this->rssObject;
 71:             } else {
 72:                 return false;
 73:             }
 74:         }
 75:       
 76:         protected function parse($data) {
 77:             // set up XML parser
 78:             $parser = xml_parser_create();
 79:             xml_parser_set_option($parser, XML_OPTION_TARGET_ENCODING, $this->outputEncoding);
 80:             // set scope of handler functions to this class
 81:             xml_set_object($parser, $this);
 82:             // set handler functions
 83:             xml_set_element_handler($parser, 'start_tag', 'close_tag');
 84:             xml_set_character_data_handler($parser, 'tag_content');
 85:             // parse the data, set flag to indicate success or failure
 86:             $result = xml_parse($parser, $data);
 87:             // free memory used
 88:             xml_parser_free($parser);
 89:          
 90:             return $result;
 91:         }
 92:       
 93:         // this function triggers each time the parser encounters a new tag
 94:         protected function start_tag($parser, $name, $attributes) {
 95:             if ($this->inItem || ($this->inChannel && $name != 'ITEM')) {
 96:                 $this->tag = $name;
 97:             } else {
 98:                 switch ($name) {
 99:                     case 'ITEM':
100:                         $this->inItem = true;
101:                         $this->pointer++;
102:                   
103:                         if ($this->numItems == -1 || $this->pointer < $this->numItems) {
104:                             $this->rssObject->items[] = new RssItem();
105:                         }
106:                         break;
107:                     case 'CHANNEL':
108:                         $this->inChannel = true;
109:                         break;
110:                 }
111:             }
112:         }
113:       
114:         // this function triggers when the parser encounters a corresponding close tag
115:         protected function close_tag($parser, $name) {
116:             if ($name == 'ITEM') {
117:                 if ($this->numItems == -1 || $this->pointer < $this->numItems) {
118:                     // if the feed contained a content element we'll override the description text with it as it's 
119:                     // likely to represent the entire article
120:                     if ($this->tempContent != '') {
121:                         $this->rssObject->items[$this->pointer]->description = $this->tempContent;
122:                         $this->tempContent = '';
123:                     }
124:                 }
125:                 $this->inItem = false;
126:             } elseif ($name == 'CHANNEL') {
127:                 $this->inChannel = false;
128:             }
129:         }

132:         protected function tag_content($parser, $data) {

142:                             break;
143:                         case 'CONTENT:ENCODED':
144:                             $this->tempContent .= $data;
145:                             break;
146:                         case 'LINK':
147:                             $this->rssObject->items[$this->pointer]->link .= $data;
148:                             break;
149:                         case 'PUBDATE':
150:                         case 'DC:DATE':
151:                             $this->rssObject->items[$this->pointer]->date .= $data;
152:                             break;
153:                         case 'AUTHOR':
154:                         case 'DC:CREATOR':

156:                             break;
157:                         case 'CATEGORY':
