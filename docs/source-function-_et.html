138:         /**
139:          * Processes selected translation file
140:          *
141:          * @return void
142:          * @author Ed Eliot
143:          **/
144:         protected function process() {
145:             // create array for serialising
146:             $cache = array();
147:             
148:             // read translation file into array
149:             $file = file($this->file);
150:             
151:             // does this translation file inherit from another - only one level of inheritance 
152:             // supported at the moment
153:             $inheritsMatches = array();
154:             if (isset($file[0]) && preg_match("/^\s*{inherits\s+([^}]+)}.*$/", $file[0], $inheritsMatches)) {
155:                 $parentFile = $this->translationsDir.trim($inheritsMatches[1]).'.txt';
156:                 // read parent file into array
157:                 $parentFile = file($parentFile);
158:                 // merge lines from parent file into main file array, lines in the main file override lines in the parent
159:                 $file = array_merge($parentFile, $file);
160:                 // store filename of parent
161:                 $cache['parent-filename'] = $parentFile;
162:                 // store timestamp of parent
163:                 $cache['parent-timestamp'] = filemtime($parentFile);
164:             }
165:             
166:             // read language array line by line
167:             foreach ($file as $line) {
168:                 $translationMatches = array();
169:                 
170:                 // match valid translations, strip comments - both on their own lines and at the end of a translation
171:                 // literal hashes (#) should be escaped with a backslash
172:                 if (preg_match("/^\s*([0-9a-z\._-]+)\s*=\s*((\\\\#|[^#])*).*$/iu", $line, $translationMatches)) {
173:                     $this->translations[$translationMatches[1]] = trim(str_replace('\#', '#', $translationMatches[2]));
174:                 }
175:             }
176:             // add current timestamp of translation file
177:             $cache['timestamp'] = filemtime($this->file);
178:             // add translations
179:             $cache['translations'] = $this->translations;
180:             // write cache
181:             file_put_contents($this->cacheFile, serialize($cache));
182:         }

198:                 $numArgs = func_num_args();
199:                 if ($numArgs > 1) { // complex translation, substitution values to process

201:                     if (is_array($firstArg)) { // named substitution variables
202:                         foreach ($firstArg as $key => $value) {
203:                             $translation = str_replace('{'.$key.'}', $value, $translation);
204:                         }
205:                     } else { // numbered substitution variables
206:                         for ($i = 1; $i < $numArgs; $i++) {
207:                             $param = func_get_arg($i);
208:                             // replace current substitution marker with value
