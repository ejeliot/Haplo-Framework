 14:      * @author Ed Eliot
 15:      * @package HaploFileCache
 16:      **/
 17:     class HaploFileCache {
 18:         protected $file;
 19:         protected $fileLock;
 20:         protected $cacheTime;
 21:         protected $shortKey;
 22:         protected static $cache = array();
 23: 
 24:         /**
 25:          * Constructor for class
 26:          *
 27:          * @param string $key Key for cached data to look up/set
 28:          * @param integer $cacheTime Length of time to cache data for (in seconds)
 29:          * @param string $cachePath File path for location of cache files
 30:          * @return void
 31:          * @author Ed Eliot
 32:          **/
 33:         public function __construct($key, $cacheTime) {
 34:             global $config;
 35:             
 36:             $this->shortKey = 'HAPLO-CACHE-'.md5($key);
 37:             $this->file = $config->get_key('paths', 'cache')."/$this->shortKey.txt";
 38:             $this->fileLock = "$this->file.lock";
 39:             $this->cacheTime = $cacheTime;
 40:         }
 41: 
 42:         /**
 43:          * Checks if cache item exists and content isn't stale
 44:          *
 45:          * @return boolean
 46:          * @author Ed Eliot
 47:          **/
 48:         public function check() {
 49:             if (array_key_exists($this->shortKey, self::$cache) || file_exists($this->fileLock)) {
 50:                 return true;
 51:             }
 52:             return (file_exists($this->file) && ($this->cacheTime == -1 || time() - filemtime($this->file) <= $this->cacheTime));
 53:         }
 54: 
 55:         /**
 56:          * Checks if cache item exists but isn't concerned if content is stale
 57:          *
 58:          * @return boolean
 59:          * @author Ed Eliot
 60:          **/
 61:         public function exists() {
 62:             return (array_key_exists($this->shortKey, self::$cache)) || (file_exists($this->file) || file_exists($this->fileLock));
 63:         }
 64: 
 65:         /**
 66:          * Cache some content
 67:          *
 68:          * @param variant $contents Data to cache
 69:          * @return boolean
 70:          * @author Ed Eliot
 71:          **/
 72:         public function set($contents) {
 73:             if (!file_exists($this->fileLock)) {
 74:                 if (file_exists($this->file)) {
 75:                     copy($this->file, $this->fileLock);
 76:                 }
 77:                 $file = fopen($this->file, 'w');
 78:                 fwrite($file, serialize($contents));
 79:                 fclose($file);
 80:                 if (file_exists($this->fileLock)) {
 81:                     unlink($this->fileLock);
 82:                 }
 83:                 self::$cache[$this->shortKey] = $contents;
 84:                 return true;
 85:             }     
 86:             return false;
 87:         }
 88: 
 89:         /**
 90:          * Get the contents of the cache
 91:          *
 92:          * @return variant
 93:          * @author Ed Eliot
 94:          **/
 95:         public function get() {
 96:             if ($this->exists()) {
 97:                 if (array_key_exists($this->shortKey, self::$cache)) {
 98:                     return self::$cache[$this->shortKey];
 99:                 } else if (file_exists($this->fileLock)) {
100:                     self::$cache[$this->shortKey] = unserialize(file_get_contents($this->fileLock));
101:                     return self::$cache[$this->shortKey];
102:                 } else {
103:                     self::$cache[$this->shortKey] = unserialize(file_get_contents($this->file));
104:                     return self::$cache[$this->shortKey];
105:                 }
106:             }
107:             return false;
108: 
109:         }
